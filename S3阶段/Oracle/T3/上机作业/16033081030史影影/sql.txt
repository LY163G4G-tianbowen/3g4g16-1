
　1、列出至少有一个员工的所有部门。
select e.deptno,count(deptno) from emp e group by e.deptno having count(deptno)>=1;

　　2、列出薪金比“SMITH”多的所有员工。
select e.ename from emp e where e.sal>(select e.sal from emp e where e.ename='SMITH');

　　3、列出所有员工的姓名及其直接上级的姓名。
select e.ename,s.ename from emp e,emp s where e.mgr=s.empno

　　4、列出受雇日期早于其直接上级的所有员工。
select e.ename from emp e, emp m where e.mgr = m.empno and e.hiredate > m.hiredate; 


　　5、列出部门名称和这些部门的员工信息，包括那些没有员工的部门。
select d.*,e.* from dept d left join emp e on d.deptno=e.deptno;

　　6、列出所有job为“CLERK”（办事员）的姓名及其部门名称。
select e.ename,d.dname from emp e left join dept d on e.deptno=d.deptno where e.JOB='CLERK';

　　7、列出最低薪金大于1500的各种工作。
select distinct(e.JOB) from emp e where e.sal>1500;

　　8、列出在部门“SALES”（销售部）工作的员工的姓名，假定不知道销售部的部门编号。
select e.ename from emp e where e.deptno=(select d.deptno from dept d where d.dname='SALES');
　　9、列出薪金高于公司平均工资的所有员工。
select e.ename from emp e where e.sal>(select avg(e.sal) from emp e);

　　10、列出与“SCOTT”从事相同工作的所有员工。
select e.ename from emp e where e.deptno=(select e.deptno from emp e where e.ename='SCOTT') and e.ename!='SCOTT';

　　11、列出薪金等于部门30中员工的薪金的所有员工的姓名和薪金。
select e.ename,e.sal from emp e where e.sal in (select e.sal from emp e where e.deptno=30) and e.deptno!=30;

　　12、列出薪金高于在部门30工作的所有员工的薪金的员工姓名和薪金。
select e.ename,e.sal from emp e where e.sal > (select max(e.sal) from emp e where e.deptno=30);

　　13、列出在每个部门工作的员工数量、平均工资和平均服务期限。

select count(e.deptno),avg(e.sal),avg(round(months_between(sysdate,e.hiredate)/12, 0)) from emp e group by e.deptno;

　　14、列出所有员工的姓名、部门名称和工资。
select e.ename,d.dname,e.sal from emp e left join dept d on d.deptno=e.deptno;

　　15、列出从事同一种工作但属于不同部门的员工的一种组合。

select e.* from emp e, emp m where e.job = m.job and e.deptno <> m.deptno;

　　16、列出所有部门的详细信息和部门人数。

select d.dname,d.deptno,d.loc, count(d.deptno) from emp e, dept d where e.deptno = d.deptno group by d.dname, d.deptno, d.loc  

　　17、列出各种工作的最低工资。
select e.job, min(e.sal) from emp e,dept d group by e.job;

　　18、列出各个部门的MANAGER（经理）的最低薪金（job为MANAGER）。
 select min(sal) from (select sal from emp where job = 'MANAGER');


　　19、列出所有员工的年工资，按年薪从低到高排序。
select e.sal*12 from emp e order by e.sal*12 asc;
